So, we already made our two language files, lang.en-UK.json and lang.el-GR.json.
Now, in our app, we include the 'interpreter' service. 

var myApp = angular.module('myApp', []);

myApp.controller('myCtrl', ['$scope', 'interpreter', function($scope, interpreter){

	// so now interpreter is available
	// let's add english ( and when file is downloaded, make it active )
	interpreter.addLang('en-UK', 'languages/lang.en-UK.json', function(){
		interpreter.setLang('en-UK');
	});
	// let's also get greek
	interpreter.addLang('el-GR', 'languages/lang.el-GR.json');

	// we can bind a local variable to the available keyword map
	$scope.keyword = interpreter.map;

	// that's it, done
	// if we want to switch to greek, we just do: interpreter.setLang('el-GR');
	// that's what the buttons below do
}]);
